
SELECT ( '*' | column_family_name [, column_family_name]* )
    FROM table_name
    [where_clause]
    [into_file_clause]

where_clause:
    WHERE predicate [&& predicate]*

predicate:
    ROW = row_key |
    START_ROW [INCLUSIVE|EXCLUSIVE] = row_key |
    END_ROW [INCLUSIVE|EXCLUSIVE] = row_key |
    MAX_VERSIONS = version_count |
    START_TIME = timestamp |
    END_TIME = timestamp |
    LIMIT = row_count

into_file_clause:
    INTO FILE 'file_name'
